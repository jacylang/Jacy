73 inline Option<PrimTypeSet> getPrimTypeBitMask(span::Symbol typeName) {
74 const auto primType = getPrimType(typeName);

83 inline std::vector<std::string> getShadowedPrimTypes(PrimTypeSet mask) {
84 static const std::map<PrimType, std::string> primTypesNames = {
85 {PrimType::Bool, "bool"},
86 {PrimType::Int, "int"},
87 {PrimType::Uint, "uint"},
88 {PrimType::I8, "i8"},
89 {PrimType::I16, "i16"},
90 {PrimType::I32, "i32"},
91 {PrimType::I64, "i64"},
92 {PrimType::U8, "u8"},
93 {PrimType::U16, "u16"},
94 {PrimType::U32, "u32"},
95 {PrimType::U64, "u64"},
96 {PrimType::Char, "char"},
97 {PrimType::Str, "str"},
98 };
99 static const auto leftmostBit = static_cast<PrimTypeSet>(PrimType::Str);
100
101 std::vector<std::string> shadowedTypes;
102 for (PrimTypeSet shift = 0; shift < leftmostBit; shift++) {
103 if (mask << shift & 1) {
104 shadowedTypes.emplace_back(primTypesNames.at(static_cast<PrimType>(shift)));
105 }
106 }
107 return shadowedTypes;
108 }
